 ## Specify function to compute SE for simple slopes:
 getSE <- function(z, lmOut) {
     tmp <- vcov(lmOut)
     varB1 <- tmp[2, 2]
     varB3 <- tmp[4, 4]
     covB13 <- tmp[4, 2]
 
     sqrt(varB1 + 2 * z * covB13 + z^2 * varB3)
 }
